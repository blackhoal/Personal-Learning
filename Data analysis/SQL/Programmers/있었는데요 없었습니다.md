# 작성 답안
~~~mysql
SELECT A.ANIMAL_ID, A.NAME 
FROM ANIMAL_INS AS A 
LEFT JOIN ANIMAL_OUTS AS B 
ON A.NAME = B.NAME
WHERE A.DATETIME > B.DATETIME
ORDER BY A.DATETIME;
~~~

# 풀이
~~~mysql
SELECT A.ANIMAL_ID, A.NAME 
FROM ANIMAL_INS AS A 
LEFT JOIN ANIMAL_OUTS AS B 
ON A.ANIMAL_ID = B.ANIMAL_ID # 
WHERE A.DATETIME > B.DATETIME
ORDER BY A.DATETIME;
~~~
# 필기
~~~
- on 구문을 A.NAME = B.NAME이 아니라 A.ANIMAL_ID = B.ANIMAL_ID로 풀이 필요
~~~
