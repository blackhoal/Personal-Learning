# 2. 변수 입력과 연산자
## ① 변수 입력
~~~python
a, b = input("입력 : ").split() # 현재 a와 b는 string 타입
print(a+b)
> 입력 : 2 3
> 23 # 현재 a와 b는 string 타입
a, b = map(int, input("입력 : ").split()) # a와 b를 int 타입으로 변환
print(a+b)
> 입력 : 2 3
> 5
~~~
## ② 연산자
~~~python
a, b = int(5), int(2)
a + b   # 덧셈
> 7
a - b   # 뺄셈
> 3
a * b   # 곱셈
> 10
a / b   # 나눗셈
> 2.5
a // b  # 나눗셈의 몫
> 2
a % b   # 나눗셈의 나머지
> 1
a ** b  # 거듭제곱
> 25
~~~
