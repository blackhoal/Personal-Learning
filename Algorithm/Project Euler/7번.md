# 문제
~~~
소수를 크기 순으로 나열하면 2, 3, 5, 7, 11, 13, ... 과 같이 됩니다.

이 때 10,001번째의 소수를 구하세요.
~~~
# 작성 답안
~~~python
num_7 = 1
list_7 = list()
while True:
    flag_7 = True
    num_7 += 1
    for i in range(2, num_7):
        if num_7 % i == 0:
            flag_7 = False
            break
            
    if flag_7:
        list_7.append(num_7)
        
    if len(list_7) == 10001:
        print(list_7[10000])
        break
~~~ 
# 타인이 작성한 답안
~~~python
result = [2]
a = 3

while True:
    count = 0
    for i in range(1, a, 2):
        if a % i == 0:
            count += 1
    if count == 1:
        result.append(a)
    if len(result) == 10001:
        break
    a += 2

print(max(result))
~~~
