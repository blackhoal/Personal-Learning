# 문제
~~~
10 이하의 소수를 모두 더하면 2 + 3 + 5 + 7 = 17 이 됩니다.

이백만(2,000,000) 이하 소수의 합은 얼마입니까?
~~~
# 작성 답안
~~~python
result = 0
for i in range(2, 2000000):
    count = 0
    for j in range(1, i):
        if i%j == 0:
            count += 1
    if count == 1 :
        result += i

result
~~~
# 타인 작성 답안
~~~python
prime = []
for check in range(2, 2000000):
    if check % 10000 == 0: 
        print(check, len(prime))
    ask = True
    for div in prime:
        if check % div == 0:
            ask = False
            break
        if div ** 2 > check:
            break
    if ask:
        prime.append(check)
print(sum(prime))
~~~

# 에라토스테네스의 체
~~~python
n=1000
a = [False,False] + [True]*(n-1)

primes=[]

for i in range(2,n+1):
  if a[i]:
    primes.append(i)
    for j in range(2*i, n+1, i):
        a[j] = False
print(primes)
~~~
