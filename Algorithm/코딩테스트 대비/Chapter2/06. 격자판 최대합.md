n = int(input())
array = [list(map(int, input().split())) for _ in range(n)]

max_value = 0
for i in range(n):
    a = b = 0
    for j in range(n):
        a += array[i][j]
        b += array[j][i]
    if max_value < a:
        max_value = a
    if max_value < b:
        max_value = b

a, b = 0, 0
for i in range(n):
    a += array[i][i]
    b += array[i][n-i-1]
    
if max_value < a:
    max_value = a
if max_value < b:
    max_value = b
    
print(max_value)
~~~
