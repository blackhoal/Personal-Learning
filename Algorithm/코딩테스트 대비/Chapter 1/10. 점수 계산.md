# 코드
~~~python
# 본인 작성 답안
n = int(input())
k = list(map(int, input().split()))

sum_10 = sum(k)
cnt = [0] * n
for i in range(1, n): # 10
    if k[i-1] == 1 and k[i] == 1: # 0 1
        cnt[i] = cnt[i-1] + 1
        
print(sum_10 + sum(cnt))
~~~

~~~python
# 강사님 작성 답안
n = int(input())
k = list(map(int, input().split()))
sum = 0
cnt = 0

for x in k:
    if x==1:
        cnt+=1
        sum+=cnt
    else:
        cnt = 0

print(sum)
~~~
