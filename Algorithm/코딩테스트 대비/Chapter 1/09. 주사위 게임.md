# 코드
~~~python
# 본인 답안
n = int(input())
data = []
for i in range(n):
    data.append(input().split())

for i in range(len(data)):
    data[i] = list(map(int, data[i]))

res = [] # 결과값
for i in range(len(data)): # 3 3 6 / 2 2 2 / 6 2 5 
    max_09 = 0 # 최대 중복 횟수
    num_09 = 0 # 중복되는 수    
    for j in range(len(data[i])): # 6 / 2 / 5
        a = data[i].count(data[i][j])
        if max_09 < a:
            max_09 = a
            num_09 = data[i][j]
    if max_09 == 1:
        num_09 = max(data[i])
        res.append(num_09 * 100)
    elif max_09 == 2:
        res.append(1000 + num_09 * 100)
    elif max_09 == 3:
        res.append(10000 + num_09 * 1000)

print(max(res))
~~~

# 강사님 답안
n = int(input())
res = 0
for i in range(n):
        tmp = input().split()
        tmp.sort()
        a, b, c = map(int, tmp)
        if a==b and b==c:
            money = 10000 + a*1000
        elif a==b or a==c:
            money = 1000 + a*100
        elif b==c:
            money = 1000 + b*100
        else:
            money = c * 100
        if money>res:
            res = money
            
print(res)
