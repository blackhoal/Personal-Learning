# 1978 소수 찾기
~~~python
n = int(input())
arr = list(map(int, input().split()))
cnt = 0

for i in arr : # 1 3 5 7
    check = 0
    if i <= 1 :
        pass
    elif i == 2 :
        cnt += 1
    else :
        for j in range(2, i):
            if i % j == 0 :
                check = 1
        if check != 1 :
            cnt += 1
    
print(cnt)
~~~
