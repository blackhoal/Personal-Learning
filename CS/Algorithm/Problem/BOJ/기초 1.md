# 2884번 알람 시계
[문제]https://www.acmicpc.net/problem/2884
~~~python
input_data = input().split()

a = int(input_data[0])
b = int(input_data[1])

b -= 45

if b < 0 :
    b += 60
    a -= 1
    if a < 0 :
        a = 23

print(str(a) + " " + str(b))
~~~

# 1110번 더하기 사이클
[문제]https://www.acmicpc.net/problem/1110
~~~python
num = temp = int(input())
count = 0
while True:
    ten = num // 10
    one = num % 10
    total = ten + one
    count += 1
    num = int(str(num % 10) + str(total % 10))
    if(temp == num):
        break
print(count) 
~~~

# 8958번 OX 
[문제]https://www.acmicpc.net/problem/8958
~~~python
# 8958
n = int(input())

for i in range(n):
    ox = input()
    score = 0
    cnt = 0
    for j in ox:
        if j == 'O':
            cnt += 1
            score += cnt
        else:
            cnt = 0
    print(score)
~~~
