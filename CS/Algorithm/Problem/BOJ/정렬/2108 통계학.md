# 문제
[2108번](https://www.acmicpc.net/problem/2108)

# 코드
~~~python
import sys
from collections import Counter

n = int(sys.stdin.readline())
arr = []
for _ in range(n):
    arr.append(int(sys.stdin.readline())) 
            
def arith_mean(arr) : 
    result = round(sum(arr) / n)
    
    return result
    
def center_value(arr) :
    result = arr[n//2]
    
    return result
        
def mode(x):
    mode_dict = Counter(x)
    modes = mode_dict.most_common()
    if len(x) > 1 : 
        if modes[0][1] == modes[1][1]:
            mod = modes[1][0]
        else : 
            mod = modes[0][0]
    else : 
        mod = modes[0][0]
    
    return mod

def ran_value(arr) :
    result = arr[-1] - arr[0]
    
    return result

print(arith_mean(arr))
arr.sort()
print(center_value(arr))
print(mode(arr))
print(ran_value(arr))
~~~
