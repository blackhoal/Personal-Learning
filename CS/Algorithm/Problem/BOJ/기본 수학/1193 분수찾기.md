# 문제
https://www.acmicpc.net/problem/1193

# 코드
~~~python
n = int(input())
floor = 1
last = (1/2) * floor * (floor + 1)

while n > last : # 14
    floor += 1
    last = (1/2) * floor * (floor + 1)

first = last - (floor -1)
arr = list(range(int(first), int(last)+1))
loc = arr.index(n)

m = loc + 1
s = floor + 1 - m
if floor % 2 == 0 :
    print(f'{m}/{s}')
else : 
    print(f'{s}/{m}')
~~~
